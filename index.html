-- Team Hub — Supabase additions -- Caring Companions In-Home Senior Care -- -- Run this in the SAME Supabase project as Care Coordinator's Hub and Staffing Hub -- (SQL Editor → New query → paste → Run). Reuses the existing `app_data` table and -- the `upsert_app_data_item` / `delete_app_data_item` RPCs from -- Supabase_Atomic_Write_Functions.sql — nothing new to create there, this script -- just seeds two more keys. Idempotent: safe to run even if app_data already exists. -- Make sure app_data exists (no-op if the other hubs already created it) create table if not exists app_data ( key text primary key, data jsonb not null default '[]'::jsonb, updated_at timestamptz not null default now() ); alter table app_data enable row level security; drop policy if exists "Authenticated read/write" on app_data; create policy "Authenticated read/write" on app_data for all using (auth.role() = 'authenticated') with check (auth.role() = 'authenticated'); -- Seed the keys this hub uses. insert into app_data (key, data) values ('standup_notes', '[]'::jsonb), ('team_meetings', '[]'::jsonb), ('team_directory', '[]'::jsonb), ('team_hub_settings', '[]'::jsonb) on conflict (key) do nothing; -- Auto-update `updated_at` on every write (matches the other two hubs) create or replace function touch_app_data_updated_at() returns trigger as $$ begin new.updated_at = now(); return new; end; $$ language plpgsql; drop trigger if exists trg_touch_app_data on app_data; create trigger trg_touch_app_data before update on app_data for each row execute function touch_app_data_updated_at(); -- Make sure Team Hub can call the shared atomic RPCs (no-op if already granted -- when you ran Supabase_Atomic_Write_Functions.sql for the other hubs). grant execute on function upsert_app_data_item(text, jsonb) to authenticated; grant execute on function delete_app_data_item(text, text) to authenticated; -- --------------------------------------------------------------------------- -- submit_standup_note_public(...) -- -- Lets the no-login Quick-Add form (quick-add.html) post a stand-up item -- WITHOUT a Supabase session. Everything else in this project requires -- auth.role() = 'authenticated' — this is the one deliberate exception, and -- it's scoped as narrowly as possible: -- - SECURITY DEFINER means it runs with the function owner's privileges, -- bypassing RLS internally — but the function body only ever does one -- thing: append one well-formed item to the standup_notes array. -- - It cannot read existing data, update an existing item, delete anything, -- or touch any other app_data key. There's no way to use this function -- to do anything except add a new stand-up note. -- - Granted to the `anon` role specifically — not `authenticated` — since -- the whole point is that no login is required. -- -- Trade-off worth knowing: anyone with the quick-add.html URL can post an -- item without identifying themselves beyond whatever they type in -- "reported by." Treat that link as semi-private (don't post it publicly) — -- share it directly with staff to bookmark. Since it can only ever add items -- (never remove or edit), the worst case is a junk entry the team can just -- delete from Team Hub once signed in. -- --------------------------------------------------------------------------- create or replace function submit_standup_note_public( p_summary text, p_category text default 'General', p_related_to text default '', p_reported_by text default '' ) returns jsonb as $$ declare new_item jsonb; begin if p_summary is null or trim(p_summary) = '' then raise exception 'summary is required'; end if; new_item := jsonb_build_object( 'id', 'id_' || substr(md5(random()::text || clock_timestamp()::text), 1, 12), 'summary', left(trim(p_summary), 2000), 'category', coalesce(nullif(trim(p_category), ''), 'General'), 'related_to', left(trim(coalesce(p_related_to, '')), 200), 'occurred_at', to_char(now() at time zone 'utc', 'YYYY-MM-DD"T"HH24:MI:SS"Z"'), 'reported_by', left(trim(coalesce(p_reported_by, '')), 100), 'resolved', false, 'resolved_by', '', 'resolved_at', '', 'created_at', to_char(now() at time zone 'utc', 'YYYY-MM-DD"T"HH24:MI:SS"Z"'), 'source', 'quick_add_form' ); insert into app_data (key, data) values ('standup_notes', '[]'::jsonb) on conflict (key) do nothing; update app_data set data = coalesce(data, '[]'::jsonb) || jsonb_build_array(new_item) where key = 'standup_notes'; return new_item; end; $$ language plpgsql security definer set search_path = public; grant execute on function submit_standup_note_public(text, text, text, text) to anon; -- --------------------------------------------------------------------------- -- Record shapes (enforced in the app's JS, not by Postgres) -- --------------------------------------------------------------------------- -- -- standup_notes[] : -- { id, summary, category ("Caregiver / Scheduling"|"Client"|"Incident"|"General"), -- related_to, occurred_at (ISO timestamp of when it actually happened), -- reported_by, resolved (bool), resolved_by, resolved_at, created_at } -- Read/written by Team Hub only. Both care coordinators and staffing coordinators -- see the same list — keep entries to operational facts, not detailed client -- health information, since visibility here is broader than either hub alone. -- -- team_meetings[] : -- { id, meeting_date, meeting_name, attendees, notes, action_items, created_at } -- One row per logged meeting. `action_items` is free text, one item per line. -- -- team_directory[] : -- { id, name, role_key, role_title, reports_to, phone, email, -- status ("in"|"on_call"|"out"), location (""|"office"|"remote"|"home"), -- sort_order, status_updated_by, status_updated_at, created_at, updated_at } -- role_key is one of: owner, manager, coordinator, staffing, lead, caregiver, other — -- it drives the card's position in the reporting chain and its avatar color. -- role_title is the free-text label actually shown on the card (defaults to the -- role_key's standard title but can be customized, e.g. a specific person's name -- for their title). Read/written through the same upsert_app_data_item / -- delete_app_data_item RPCs as everything else — no new function needed. Adding, -- editing, or removing a person is gated client-side behind the admin passcode -- (owners only); updating `status`/`location` on an existing person is open to -- anyone signed into Team Hub, by design — it's meant to work like a physical -- in/out board. -- -- team_hub_settings[] : -- Singleton-style records (always looked up by fixed `id`, never a growing list). -- Currently holds one record: -- { id: "video_room", room_name, created_at } -- room_name is a long, unguessable slug used as the Jitsi Meet room for the -- "Team Video Room" feature — https://meet.jit.si/. It's created -- automatically the first time anyone clicks "Start / Join Video Meeting" (no -- setup required), and syncs here so every team member's browser resolves to -- the same room. Jitsi's free public server has no login and no access control -- beyond knowing the room name, so treat this value the way you'd treat a -- shared Zoom link — don't post it somewhere public. Owners can rotate it from -- Team Hub → Settings → Video Meeting Room if it's ever been over-shared.
